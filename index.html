    const foodMenu = document.getElementById("foodMenu");

    const drinkMenu = document.getElementById("drinkMenu");

    const orderItemsDiv = document.getElementById("orderItems");

    const totalDiv = document.getElementById("total");

    const balanceDiv = document.getElementById("balance");

    const receiptDiv = document.getElementById("receipt");

    let order = [];


    menuItems.forEach(item => {

      const btn = document.createElement("button");

      btn.textContent = `${item.name}\nRM${item.price.toFixed(2)}`;

      btn.onclick = () => addItem(item);

      (item.type === "makanan" ? foodMenu : drinkMenu).appendChild(btn);

    });


    function addItem(item) {

      order.push(item);

      renderOrder();

    }


    function renderOrder() {

      orderItemsDiv.innerHTML = "";

      let total = 0;

      order.forEach((item, index) => {

        total += item.price;

        const div = document.createElement("div");

        div.textContent = `${item.name} - RM${item.price.toFixed(2)}`;

        orderItemsDiv.appendChild(div);

      });

      totalDiv.textContent = `Total: RM${total.toFixed(2)}`;

      balanceDiv.textContent = `Balance: RM0.00`;

    }


    function finalizeOrder(type) {

      if (order.length === 0) return alert("Add items first!");

      const total = order.reduce((sum, item) => sum + item.price, 0);

      let receipt = "🧾 Resit Sejemput Stall POS\n--------------------------\n";

      receipt += `Tarikh: ${new Date().toLocaleString()}\n\n`;

      order.forEach(item => {

        receipt += `${item.name} - RM${item.price.toFixed(2)}\n`;

      });

      receipt += "--------------------------\n";


      if (type === 'pay') {

        const paid = parseFloat(prompt(`Total is RM${total.toFixed(2)}. Enter amount paid:`));

        if (isNaN(paid) || paid < total) return alert("Not enough payment!");

        const balance = paid - total;

        balanceDiv.textContent = `Balance: RM${balance.toFixed(2)}`;

        receipt += `Total: RM${total.toFixed(2)}\nDibayar: RM${paid.toFixed(2)}\nBaki: RM${balance.toFixed(2)}\nBayar: ${document.getElementById("paymentType").value.toUpperCase()}\n`;

      } else {

        receipt += `Payment: LATER\n`;

      }

      receipt += "--------------------------\nTerima kasih!";

      receiptDiv.textContent = receipt;

    }


    function clearOrder() {

      order = [];

      receiptDiv.textContent = "";

      renderOrder();

    }
